#include <bits/stdc++.h>
using namespace std;
using i64 = long long;
#define endl "\n"

string m = "NSEW";

int main()
{
  i64 T;
  cin >> T;
  for (i64 _ = 1; _ <= T; _++)
  {
    i64 X, Y;
    cin >> X >> Y;
    for (i64 i = 1; i <= 10; i++)
    {
      for (i64 j = 0; j < pow(4, i); j++)
      {
        string ans;
        i64 tmp = j;
        for (i64 k = 0; k < i; k++)
        {
          ans += m[tmp % 4];
          tmp /= 4;
        }
        i64 nowX = 0, nowY = 0;
        for (i64 k = 0; k < i; k++)
        {
          i64 move = 1LL << k;
          if (ans[k] == 'E')
            nowX += 1LL << k;
          else if (ans[k] == 'W')
            nowX -= 1LL << k;
          else if (ans[k] == 'N')
            nowY += 1LL << k;
          else if (ans[k] == 'S')
            nowY -= 1LL << k;
        }
        if (nowX == X && nowY == Y)
        {
          cout << "Case #" << _ << ": " << ans << endl;
          goto ok;
        }
      }
    }
    cout << "Case #" << _ << ": "
         << "IMPOSSIBLE" << endl;
  ok:;
  }
  return 0;
}
